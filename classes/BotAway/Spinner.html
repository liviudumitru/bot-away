  def initialize(ip, key, secret)
    raise "Shouldn't have a nil ip" unless ip
    raise "Shouldn't have a nil secret" unless secret
    secret = File.join(#Time.now.to_i.to_s,
                                               ip,
                                               key.to_s,
                                               secret)

    @spinner = Digest::MD5.hexdigest(secret)
  end
